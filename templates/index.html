Quick Example

Check if a user is in a Roblox group

curl -X GET "https://arrow-api.up.railway.app/123456789/987654321" \
  -H "api-key: your-api-key-here"
const response = await fetch('https://arrow-api.up.railway.app/123456789/987654321', {
  headers: {
    'api-key': 'your-api-key-here'
  }
});
const data = await response.json();
import requests

response = requests.get(
    'https://arrow-api.up.railway.app/123456789/987654321',
    headers={'api-key': 'your-api-key-here'}
)
data = response.json()
